STEP 4 — FIX CHECKOUT PRICE MAPPING
Find where checkout sessions are created and map Stripe price IDs by plan + billing period.

Use a helper like this:

```ts
const getStripePriceId = (
  tier: string,
  billingPeriod: 'monthly' | 'annual'
) => {
  if (tier === 'local-launch') {
    return billingPeriod === 'annual'
      ? process.env.STRIPE_PRICE_LOCAL_LAUNCH_ANNUAL
      : process.env.STRIPE_PRICE_LOCAL_LAUNCH_MONTHLY;
  }

  if (tier === 'founding-agency') {
    return process.env.STRIPE_PRICE_FOUNDING_AGENCY_MONTHLY;
  }

  if (tier === 'growth-bundle') {
    return billingPeriod === 'annual'
      ? process.env.STRIPE_PRICE_GROWTH_BUNDLE_ANNUAL
      : process.env.STRIPE_PRICE_GROWTH_BUNDLE_MONTHLY;
  }

  return null;
};
```

Then checkout logic should:
- Use the mapped price ID
- Send billingPeriod and tier in the request
- Prevent checkout for Enterprise if it is contact-only

Example:

```ts
const createCheckout = async (tier: string) => {
  const priceId = getStripePriceId(tier, billingPeriod);

  if (!priceId) {
    if (tier === 'enterprise') {
      window.location.href = '/contact';
      return;
    }
    throw new Error(`Missing Stripe price for tier: ${tier}`);
  }

  const response = await fetch('/api/stripe/create-checkout-session', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      tier,
      billingPeriod,
      priceId,
    }),
  });

  const session = await response.json();

  if (!response.ok || !session?.url) {
    throw new Error(session?.message || 'Failed to create checkout session');
  }

  window.location.href = session.url;
};
```
