IMPLEMENTATION NOTES
Update the pricing card render logic so each plan can return billing-period-specific display strings.

Suggested pattern:

```ts
const getPlanDisplay = (plan: any, billingPeriod: 'monthly' | 'annual') => {
  if (plan.slug === 'local-launch') {
    if (billingPeriod === 'annual') {
      return {
        ctaText: 'Choose Annual',
        billingText: 'Billed annually at $19,171.20',
        savingsText: 'Save $4,792.80/year',
      };
    }

    return {
      ctaText: 'Start Free Trial',
        billingText: plan.commitmentText || '6-month commitment',
        savingsText: null,
    };
  }

  if (plan.slug === 'growth-bundle') {
    if (billingPeriod === 'annual') {
      return {
        ctaText: 'Get Started Annual',
        billingText: 'Billed annually at $28,800',
        savingsText: 'Save $7,200/year',
      };
    }

    return {
      ctaText: 'Get Started',
      billingText: plan.commitmentText || '6-month commitment',
      savingsText: null,
    };
  }

  if (plan.slug === 'founding-agency') {
    return {
      ctaText: 'Claim Offer →',
      billingText: plan.commitmentText || '6-month commitment',
      savingsText: null,
    };
  }

  if (plan.slug === 'enterprise') {
    return {
      ctaText: 'Contact Sales',
      billingText: plan.commitmentText || '6-month commitment',
      savingsText: null,
    };
  }

  return {
    ctaText: 'Get Started',
    billingText: null,
    savingsText: null,
  };
};
```
